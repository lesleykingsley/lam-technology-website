  let wordIndex = 0;
  const rotatingEl = document.getElementById('rotatingWord');

  function rotateWord() {
    // Fade out
    rotatingEl.style.transition = 'opacity 0.4s ease, transform 0.4s ease';
    rotatingEl.style.opacity = '0';
    rotatingEl.style.transform = 'translateY(-12px)';

    setTimeout(function() {
      wordIndex = (wordIndex + 1) % words.length;
      rotatingEl.textContent = words[wordIndex];
      // Fade in from below
      rotatingEl.style.transform = 'translateY(12px)';
      rotatingEl.style.opacity = '0';

      setTimeout(function() {
        rotatingEl.style.opacity = '1';
        rotatingEl.style.transform = 'translateY(0)';
      }, 50);
    }, 400);
  }

  setInterval(rotateWord, 2800);

  // ── SCROLL REVEAL ANIMATIONS ───────────────────────
  const revealObserver = new IntersectionObserver(function(entries) {
    entries.forEach(function(entry) {
      if (entry.isIntersecting) {
        entry.target.classList.add('visible');
        revealObserver.unobserve(entry.target);
      }
    });
  }, {
    threshold: 0.12,
    rootMargin: '0px 0px -40px 0px'
  });

  document.querySelectorAll('.reveal').forEach(function(el, i) {
    // Stagger cards within the same grid
    el.style.transitionDelay = (i % 6) * 0.08 + 's';
    revealObserver.observe(el);
  });

  // ── NAVBAR SCROLL EFFECT ───────────────────────────
  window.addEventListener('scroll', function() {
